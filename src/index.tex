% \begin{center}
    % \Large \textbf{Moser's Circle Problem} \\[0.2cm]
    % \large \textit{(Maximum number of regions formed by chords joining points on a circle)}
% \end{center}

\vspace{0.5cm}

% Problem description
\noindent Given $m$ points on a circle, join every pair with a chord such that no three chords intersect at a single interior point.

\vspace{0.5cm}

% Questions list
\begin{enumerate}[label=\textbf{Question \arabic*.}, leftmargin=*]
    \item If $m$ points are placed on a circle and every pair of points is joined by a chord, how many chords are drawn?
    
    \item How many intersection points are formed inside the circle when no three chords are concurrent?
    
    \item Using the results from the above parts, obtain a formula for the maximum number of regions formed inside the circle.
\end{enumerate}

\newcommand{\drawoverlay}[3]{

            \foreach \j in {\jmin,...,#1}{
                \draw (P\i) -- (P\j);
            }
        }
    \fi
    \foreach \i in {1,...,#1}{
        \node[point] at (P\i) {};
    }
    \node[below, font=\Large] at (0, -\R - 0.5) {$n=#1, c=#2, r_G=#3$};
}
