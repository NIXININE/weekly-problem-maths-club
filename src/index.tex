\begin{exer*}

\noindent A rare species of honeybee builds hives made of hexagonal cells. A complete hive consists of \textbf{37 hexagonal cells} arranged in concentric rings:

\begin{minipage}{0.6\textwidth}
    
    \begin{itemize}[noitemsep, topsep=5pt]
        \item Center: 1 hexagon
        \item $1^{\text{st}}$ Ring: 6 hexagons
        \item $2^{\text{nd}}$ Ring: 12 hexagons
        \item $3^{\text{rd}}$ Ring: 18 hexagons
    \end{itemize}
\end{minipage}%
\hfill 
\begin{minipage}{0.35\textwidth}
    \centering
    % Replace 'hexagon.png' with your file
    \includegraphics[width=\linewidth]{hexagonal_diagram.png} 
    \captionof{figure}{Hive Arrangement}
\end{minipage}

\noindent \textbf{Setup:} You are given \textbf{60 bees} to build as many complete hives as possible based on the following rules:

\begin{enumerate}
    \item \textbf{Direct Placement:} You may place a bee directly into a cell to fill it (consumes 1 bee).
    \item \textbf{Automatic Filling:} Any empty cell with \textbf{3 or more filled neighbors} fills automatically (consumes 0 bees).
    \item Automatic filling can trigger chain reactions.
\end{enumerate}

\noindent \textbf{Questions:}
\begin{enumerate}
    \item What is the \textbf{minimum} number of bees required to fill one complete hive of 37 hexagons?
    \item Using this optimal strategy, what is the \textbf{maximum} number of complete hives that can be built using 60 bees?
\end{enumerate}



\end{exer*}
